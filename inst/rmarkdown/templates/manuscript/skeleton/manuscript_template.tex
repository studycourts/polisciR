% dtholmes@mail.ubc.ca post on the lab-r-torian
% add additional author blocks as needed to YAML in MyPaper.Rmd
% use "number" field to indicate author order

$if(author)$
    \usepackage{authblk}
    $for(author)$
        $if(author.name)$
            $if(author.number)$
                \author[$author.number$]{$author.name$}
            $else$
                \author[]{$author.name$}
            $endif$
            $if(author.affiliation)$
                $if(author.email)$
                    \affil{$author.affiliation$ \thanks{$author.email$}}
                $else$
                    \affil{$author.affiliation$}
                $endif$
            $endif$
            $else$
            \author{$author$}
        $endif$
    $endfor$
$endif$
